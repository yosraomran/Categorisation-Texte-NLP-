-
حزن
في
الأراضي
الفلسطينية
:
وضربت
الأراضي
الفلسطينية
حالة
من
الحزن
والتوتر
الشديدين
في
وقت
مبكر
أمس
(
الخميس
)
فور
سماع
نبأ
وفاة
عرفات
.
وتحولت
سماء
مدن
غزة
والضفة
الغربية
إلى
غمامة
سوداء
جراء
إشعال
النار
في
إطارات
السيارات
فيما
سمع
دوي
طلقات
نارية
كثيفة
في
الشوارع
الرئيسية
وفي
محيط
المقرات
الأمنية
للسلطة
الفلسطينية
.
وأشعلت
السيارات
المارة
الأضواء
فيما
امتنع
الطلبة
والموظفون
من
التوجه
إلى
مؤسساتهم
.
وبدى
الوجوم
وعدم
التصديق
الحالة
السائدة
في
أوساط
الفلسطينيين
.
وقال
أحمد
الغراوي
:
رحمك
الله
يا
ابو
عمار
وكان
الله
في
عوننا
من
بعدك
.
وبدأت
دور
العبادة
بتلاوة
القرآن
فيما
تدفق
العشرات
من
الفلسطينيين
إلى
الشوارع
.
وفي
رام
الله
بدأ
المئات
من
الفلسطينيين
يتوافدون
إلى
مقر
المقاطعة
الذي
حوصر
به
عرفات
لمدة
تزيد
عن
عامين
ونصف
وسيدفن
فيها
وقد
أصابهم
الذهول
والحزن
.
ويرى
مراقبون
فلسطينيون
أن
غياب
عرفات
سيترك
فراغا
كبيرا
داخل
الساحة
الفلسطينية
في
ظل
ضعف
النظام
السياسي
والمؤسساتي
الفلسطيني
والافتقاد
للقيادة
الجماعية
وكان
عرفات
الرجل
الوحيد
القادر
على
الإمساك
بكافة
الخيوط
وأوراق
اللعب
فيها
بلا
منازع
.
ووضع
جميل
أبو
حمدان
الموظف
في
إحدى
دوائر
السلطة
الفلسطينية
في
غزة
كلتا
يديه
على
وجهه
وطأطأ
رأسه
تعبيرا
عن
ألمه
وقال
:
خرجت
من
منزلي
كالمجنون
.
لا
أدري
ماذا
أفعل
ولمن
أبكي
وعلى
من
.
وكشف
الشيخ
عكرمة
صبري
مفتي
القدس
وفلسطين
أنه
سيحمل
معه
حفنة
من
رمال
المسجد
الأقصى
ليضعها
فوق
جثمان
الرئيس
الفلسطيني
ياسر
عرفات
بعد
أن
رفضت
إسرائيل
السماح
بدفنه
في
باحة
المسجد
كما
أوصى
.
وتابع
:
سأحمل
بنفسي
كمية
بحجم
قبضة
اليد
من
رمال
باحة
المسجد
الأقصى
لوضعها
فوق
جثمانه
عند
الدفن
في
رام
الله
بحسب
تعبيره
.
وقال
سأحملها
بقبضتي
(
حفنة
التراب
)
من
الأرض
وأضعها
في
كيس
وأذهب
إلى
رام
الله
لأرشها
في
القبر
.
إن
هذا
لا
يضر
ولا
ينفع
لكنه
يشفي
الغليل
.
وذكر
المفتي
أنه
تم
اختياره
هو
شخصيا
ليقيم
صلاة
الميت
على
عرفات
بعد
.
وقال
:
يمكن
إقامة
أكثر
من
صلاة
على
الميت
وفقا
للشرع
على
أي
حال
.
في
إشارة
منه
إلى
إمكانية
إقامة
صلاة
الميت
على
عرفات
في
باريس
وثانية
في
القاهرة
وثالثة
في
رام
الله
.
لكنه
رغب
بأن
يقوم
هو
شخصيا
بإقامة
صلاة
الميت
على
الرئيس
الفلسطيني
في
رام
الله
.
وكشف
صبري
أنهم
أخبروه
بأن
جثمان
عرفات
لن
يدفن
مباشرة
في
القبر
بل
داخل
صندوق
من
الطوب
أي
الباطون
يوضع
بدوره
في
حفرة
القبر
وذلك
تمهيدا
لنقله
مستقبلا
إلى
باحة
المسجد
الأقصى
تنفيذا
لوصيته
وهذا
ما
نسميه
بالقبر
المتنقل
وهو
شرعي
وفق
تعبيره
.
وأعلنت
حركة
فتح
أمس
(
الخميس
)
حالة
الاستنفار
القصوى
في
أوساط
عناصر
الحركة
بعد
الإعلان
عن
وفاة
عرفات
.
وقال
مصدر
مسئول
في
الحركة
أن
جميع
الأطراف
المتصارعة
داخل
الحركة
قررت
تجاوز
خلافاتها
والالتفاف
والتوحد
والتعاون
من
اجل
إظهار
أقصى
درجات
الاحترام
لعرفات
حال
الإعلان
عن
الوفاة
.
من
جهته
أعرب
نافذ
عزام
أحد
قيادي
حركة
الجهاد
الإسلامي
عن
أسفه
لوفاة
الرئيس
الفلسطيني
ياسر
عرفات

أنها
خسارة
كبيرة

الفلسطيني

لن
تنهي
القضية
الفلسطينية
.
وأشار
إلى
أن
حركته
ستبذل
كل
ما
في
وسعها
من
أجل
وحدة
ولحمة
الصف
الفلسطيني
والحفاظ
على
تماسكه
ووحدته
بعد
موت
عرفات
ومواصلة
المسيرة
التي
كرس
عرفات
نفسه
لها
في
مواجهة
الاستعمار
والاحتلال
الاسرائيلي
.
الى
ذلك
نعت

المقاومة
الإسلامية
حماس
ببالغ
الحزن
والأسى
إلى
جماهير
الشعب
الفلسطيني
وجماهير
الأمة
العربية
والإسلامية
الزعيم

ياسر
عرفات
.
ووصفت
الحركة

بأنه
القائد
والرمز
الكبير
الرئيس
ياسر
عرفات
رئيس
السلطة
الوطنية
الفلسطينية
و

التحرير
الفلسطينية
.
وقال
البيان

(
الوطن
)

القضية
الفلسطينية
والنضال

أجلها
والدفاع
عنها

المنابر
الإقليمية
والدولية

الأذى
والحصار

.
وأكد

الفلسطيني

.
وفي

)

.
وفيما

)
